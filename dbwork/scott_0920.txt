-- comm 이 null 인 데이터 조회
select * from emp where comm is null; -- null은 아스키 코드로 0 인 값이기에 '' 없이 입력
-- comm 이 null 이 아닌 데이터 조회
select * from emp where comm is not null;
-- comm 이 null 인 경우 0으로 출력, 마지막 컬럼에 sal+comm 을 더한 값을 출력
select ename,sal,comm,sal+comm from emp; -- 숫자 + null = null 
--(NullPointException 뜨니까 자바 코드에서 예외 처리 해야 함)

-- NVL (컬럼명, 값) : 해당 컬럼 값이 null 이면 0으로 출력 (mysql 에는 isnull)
select ename,sal,NVL(comm,0),sal+NVL(comm,0) from emp; --컬럼명을 주목, 함수로 뜬다
-- 컬럼 번호는 앞에 부터 1 부터 시작하는데 주로 컬럼 번호로 가져가지 않는데 언제든지 컬럼이 추가되거나 빠질 수 있기 때문이다
-- 그래서 나중에 자바로 가져갈 경우 컬럼 인덱스나 컬럼명으로 가져간다. 
-- 컬럼명이 복잡할 경우 별칭을 지정 한다
select ename as "사원명", sal as "연봉" from emp; --"별칭" , 단 공백이 없을 경우 "" 생략
select ename as 사원명, sal as 연봉 from emp;
select empno "사원 번호", sal 연봉 from emp; --별칭에 공백이 있을 경우 반드시 "" 안에 (as는 생략)

-- GROUP BY
-- 통계를 편하게 낼 수 있게 도와줌 PPT p.59
-- job 직업별로 일단 인워 수를 구해보자

-- 위의 결과를 직업의 오름차순으로 출력
select job 직업, count(*) 인원수 from emp group by job order by 직업;

-- 인원수의 오름차순으로 출력

-- 직업별로 인원수, 평균연봉과 최저연봉, 최고연봉을 출력 해보자
select job 직업,count(*) 인원수,round(avg(sal),0) 평균연봉,max(sal) 최고연봉,
    min(sal) 최저연봉 from emp group by job;
    
    
-- 그룹에 대한 조건을 줄 경우 having
-- 위의 sql문에 인원수가 3명 이상인 그룹만 출력

    min(sal) 최저연봉 from emp group by job having count(*)>=3;
